function palindrome(str) {

    // remove non alphabet & numerical characters
    const alphaArr = str.split('').filter((char) => /[a-z0-9]/i.test(char));
    
    // create reverse
    const reverseArr = alphaArr.toReversed();
    
    // make forwards and backwards lowercase strings
    const alphaStr = alphaArr.join('').toLowerCase();
    
    const reverseStr = reverseArr.join('').toLowerCase();
    
    
    //check for equality
    return reverseStr === alphaStr;
    }

function convertToRoman(num) {

    const romanNums = [

        ["M", null, null],

        ["C", "D", "M"],

        ["X", "L", "C"],

        ["I", "V", "X"]

    ];

    

    const numArr = num.toString().split('');

    const romanArr = numArr.map(convert)

    

    function convert(digit, indexOfDigit, array) {

        let numeral = []

        const currentRomNest= romanNums.length - (array.length - indexOfDigit);

    

        for (let i = 1; i <= digit; i++) {

            if (i <= 3 || i >= 6 && i <= 8) {

                numeral.push(romanNums[currentRomNest][0])

            } else if (i === 4 || i === 9) {

                const indexCurrentLetter = romanNums[currentRomNest].indexOf(numeral[0]);

                const nextLetter = romanNums[currentRomNest][indexCurrentLetter + 1]

    

                numeral.push(nextLetter)

                numeral = numeral.slice(-2)

    

            } else if (i === 5) {

                numeral.splice(0, 1);

            }

        }

        return numeral.join('')

    }

    return romanArr.join('')
}

How does the code work?

I wanted to see if it was possible to solve this without putting every possibility for each unit in the roman numeral lookup array. Instead, each decimal unit set has three possible letters that get used to build up the full numeral set by utilising the index position of the lookup array in relation to the the index position of each digit from the number given to the function.

For each of these digits, a counter is reset and counts up to the value of the digit. A set of conditionals then build up that unit's part of the numeral (including removing sections if needed) based on the corresponding decimal unit set in the roman numeral lookup array, until the counter finishes.

Once all digits have been iterated over, the results are joined back into a string and this is returned as the result of the function.
